UIAlertController使用

UIAlertController基本可以简单的理解为UIAlertView + UIActionSheet。
需要注意的是：UIAcertController是要在viewDidAppear之后加载才会显示。

UIAlertView

 - (void)addAlertView1 {
UIAlertController *alertController1 = [UIAlertController alertControllerWithTitle:@"提示" message:@"你好" preferredStyle:UIAlertControllerStyleAlert];
UIAlertAction *okAction = [UIAlertAction actionWithTitle:@"确定" style:UIAlertActionStyleDefault handler:^(UIAlertAction * _Nonnull action) {
    NSLog(@"确定按钮被点击");
}];
UIAlertAction *canselAction = [UIAlertAction actionWithTitle:@"取消" style:UIAlertActionStyleCancel handler:^(UIAlertAction * _Nonnull action) {
    NSLog(@"取消按钮被点击");
}];

[alertController1 addAction:okAction];
[alertController1 addAction:canselAction];

[self presentViewController:alertController1 animated:YES completion:nil];
}


- (void)addAlertView2 {

UIAlertController *alertController2 = [UIAlertController alertControllerWithTitle:@"提示" message:@"hello用户" preferredStyle:UIAlertControllerStyleAlert];
[alertController2 addTextFieldWithConfigurationHandler:^(UITextField * _Nonnull textField1) {
    textField1.keyboardType = UIKeyboardTypeNumberPad;
    textField1.placeholder = @"请输入用户名";
}];
[alertController2 addTextFieldWithConfigurationHandler:^(UITextField * _Nonnull textField2) {
    textField2.keyboardType = UIKeyboardTypeNumberPad;
    textField2.placeholder = @"请输入密码";
}];
UIAlertAction *canselAction = [UIAlertAction actionWithTitle:@"取消" style:UIAlertActionStyleCancel handler:nil];
UIAlertAction *okAction = [UIAlertAction actionWithTitle:@"确定" style:UIAlertActionStyleDefault handler:^(UIAlertAction * _Nonnull action) {
    UITextField *textField1 = [[alertController textFields] firstObject];
    if (textField1.text.length == 0) {
        // 这里加入自己的逻辑
        NSLog(@"用户名不能为空");
    }
    UITextField *textField2 = [[alertController textFields] objectAtIndex:1];
    if (textField2.text.length == 0) {

        NSLog(@"密码不能为空");
    }
}];
[alertController2 addAction:canselAction];
[alertController2 addAction:okAction];
[self presentViewController:alertController2 animated:YES completion:nil];

}

ActionSheet

- (void)addActionSheet {

UIAlertController *actionSheetController = [UIAlertController alertControllerWithTitle:@"请选择" message:@"你太牛了" preferredStyle:UIAlertControllerStyleActionSheet];

UIAlertAction *save = [UIAlertAction actionWithTitle:@"save" style:UIAlertActionStyleDefault handler:^(UIAlertAction * _Nonnull action) {
    NSLog(@"选择了%@",action.title);
}];
UIAlertAction *delete = [UIAlertAction actionWithTitle:@"delete" style:UIAlertActionStyleDefault handler:^(UIAlertAction * _Nonnull action) {
    NSLog(@"选择了%@",action.title);
}];
UIAlertAction *cancle = [UIAlertAction actionWithTitle:@"cancle" style:UIAlertActionStyleDefault handler:^(UIAlertAction * _Nonnull action) {
    NSLog(@"选择了%@",action.title);
}];

[actionSheetController addAction:save];
[actionSheetController addAction:delete];
[actionSheetController addAction:cancle];

[self presentViewController:actionSheetController animated:YES completion:nil];

}
